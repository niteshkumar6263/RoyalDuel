const socket = io();
let myRole = "";

// Get assigned role from server
socket.on("roleAssigned", (role) => {
  myRole = role;
  console.log("Assigned role:", myRole);
});

// Send key press to server
window.addEventListener("keydown", (e) => {
  socket.emit("keyPress", e.key);
});

// Listen to any player's action
socket.on("playerAction", ({ role, key }) => {
  console.log(`Player ${role} pressed key: ${key}`);

  // Handle movement or attack based on role + key
  if (role === "player1") {
    // Move/attack player1 sprite
  } else if (role === "player2") {
    // Move/attack player2 sprite
  }
});
